16 std::string gen_random(const int len, char c = '\0') {
17  static const char alphanum[] =
18  "0123456789"
19  "ABCDEFGHIJKLMNOPQRSTUVWXYZ"
20  "abcdefghijklmnopqrstuvwxyz";
21 
22  std::stringstream ss;
23 
24  for (int i = 0; i < len; ++i) {
25  if (c == '\0') {
26  ss << alphanum[rand() % (sizeof(alphanum) - 1)];
27  } else {
28  ss << c;
29  }
30  }
31  return ss.str();
32 }
